% HYBRID APPROACH: Table for Title/Date, Normal Text for Description
\newcommand{\entry}[5]{
    \noindent
    % The Table: Locks the Title left and Date right
    \begin{tabularx}{\linewidth}{@{} X r @{}}
        {\bfseries\large #1} & {\color{primary}\bfseries #3} \\
        {\color{accent}\itshape #2} & {\color{accent}\small #4}
    \end{tabularx}
    \vspace{2pt} % Small gap between Title and Body
    \par         % End paragraph to switch to normal text mode
    \noindent    % Ensure body text starts flush left
    #5           % The body text (or list)
    \vspace{10pt}% Space after the entry
}
